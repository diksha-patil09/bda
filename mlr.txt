x1 <- c(1,2,3,4,5); x2 <- c(2,1,4,3,5); y <- c(2,3,5,4,5)
X <- cbind(1, x1, x2)
b <- solve(t(X) %% X) %% t(X) %*% y
cat("Multiple Regression Coefficients:\n"); print(b)
y_pred <- X %*% b
plot(y_pred, y, pch=19, col="blue", main="Multiple Linear Regression", xlab="Predicted Y", ylab="Actual Y")
abline(0,1, col="red", lwd=2)  # 45-degree reference line
